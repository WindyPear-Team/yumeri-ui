Modal

Modal Title

Click the cancel button or outside the box to close.

<!-- Trigger -->
<button class="btn primary" data-toggle="modal" data-target="#myModal">Open Modal</button>

<!-- Modal Structure -->
<div class="modal" id="myModal">
  <div class="box">
    <h3>Modal Title</h3>
    <p>...</p>
    <div class="footer">
      <!-- Use data-dismiss="modal" to close -->
      <button class="btn" data-dismiss="modal">Cancel</button>
      <button class="btn primary">OK</button>
    </div>
  </div>
</div>

<div class="notification success" id="myNotification">
  <p>This is a notification message!</p>
  <button class="btn btn-clear" data-dismiss="notification" aria-label="Close"></button>
</div>
